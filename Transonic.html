def get_trig_func(q,Gamma,xi):
    s2=2./(5.*Gamma)*((xi**2.+Gamma)-((xi**2.+Gamma)**2.-5.*Gamma*q*xi**3.)**0.5)
    c2=1.-s2
    
    return s2, c2

Now the ancillary functions $f$ and $g$ defined in Equations (67) and (68) of Adams & Gregory (2012): \begin{equation} f(\xi)=\Gamma(5\cos^2\theta-3)+2\xi^2 \end{equation} and, \begin{equation} g(\xi)=\frac{3}{4}\Gamma(5\cos^2\theta-1)+\xi^2 \end{equation}

In [154]:
def get_f_g(q,Gamma,xi):
    
    s2,c2=get_trig_func(q,Gamma,xi)
    f=Gamma*(5.*c2-3.)+2.*xi**2.
    g=3./4.*Gamma*(5.*c2-1.)+xi**2.
    
    return f,g

We also need the derivative of $f$ with respect to $\xi$, using equation (67) of Adams of Gregory (2012), with; \begin{equation} f(\xi)=2\left[\left(\xi^2+\Gamma\right)^2-5\Gamma q\xi^3\right]^{1/2} \end{equation} we find: \begin{equation} \frac{{\rm d}f}{{\rm d}\xi}=\frac{4\xi(\xi^2+\Gamma)-15\Gamma q \xi^2}{\left[\left(\xi^2+\Gamma\right)^2-5\Gamma q\xi^3\right]^{1/2}} \end{equation}

def get_df_dx(q,Gamma,xi):
    df_dx=(4.*xi*(xi**2.+Gamma)-15.*Gamma*q*xi**2.)/((xi**2.+Gamma)**2.-5.*Gamma*q*xi**3.)**0.5
    
    return df_dx

We also need the scale factor $h_p$ which from Equation (64) of Adams & Gregory (2012) is: \begin{equation} h_p=\xi^5\left[f^2\cos^2\theta+g^2\sin^2\theta\right]^{-1/2} \end{equation}

def get_hp(q,Gamma,xi):
    f,g=get_f_g(q,Gamma,xi)
    s2,c2=get_trig_func(q,Gamma,xi)
    
    hp=xi**5./np.sqrt(f**2.*c2+g**2.*s2)
    
    return hp

def get_up(q,Gamma,xi):
    f,g=get_f_g(q,Gamma,xi)
    fx=get_df_dx(q,Gamma,xi)
    
    BB=xi/(5.*Gamma)*(5*Gamma*f*fx+(g*(0.75*fx-xi)-f*fx)*(2.*xi**2.+2*Gamma-f)+(g**2.-f**2)*(2.*xi-0.5*fx))
    Y=5.-BB/(f**2.+(g**2.-f**2.)*(2.*xi**2.+2*Gamma-f)/(5.*Gamma))
    
    return Y

Now for later note that the force equation is of the form: \begin{equation} \left(u-u^{-1}\right)\frac{{\rm d}u}{{\rm d}\xi}=S(\xi) \end{equation} in order to go through the sonic point with an inflow solution ${\rm d}u/{\rm d}\xi<0$. Now since $u<1$ at a distance slightly larger than the sonic point, this means $S(\xi+\epsilon)>0$, at a distance slightly smaller than the sonic point $u>1$, so $S(\xi-\epsilon)<0$, so a transonic solution also requires that: \begin{equation} \left.\frac{{\rm d}S(\xi)}{{\rm d}\xi}\right|_{\xi_s}>0. \end{equation}

Now we can turn our attention to the full case.

In the full case, the sonic point criterion $S(\xi_s)$ can be written: \begin{equation} S'(\xi_s)=(n+\frac{1}{2})\left(\frac{b-\omega\Lambda(\xi_s)\xi_s^2}{\Upsilon(\xi_s)\xi_s}\right)-b(\xi_s^{-1}-\xi_d^{-1})-n-\omega[I(\xi_s)-I(\xi_d)]-\frac{1}{2}\left[\frac{h_p(\xi_s)}{h_p(\xi_d)}\right]^2\left(\frac{b-\omega\Lambda(\xi_s)\xi_s^2}{\Upsilon(\xi_s)\xi_s}\right)^{2n+1}=0 \end{equation} to calculate this we need to define a function that evaluates $I(b)-I(a)$.

Now remeber sonic point requires $S'(\xi_s)=0.$ and, \begin{equation} \left.\frac{{\rm d}S'}{{\rm d}\xi}\right|_{\xi_s}>0 \end{equation}

def get_full_S(b,n,q,xid,Gamma,xi):
    L=get_lambda(xi,q,Gamma) # remember get_lambda is specified in a form for the interfral with xi first!
    Y=get_up(q,Gamma,xi)
    hp=get_hp(q,Gamma,xi)
    hp_d=get_hp(q,Gamma,xid)
    w=get_w(xid,b)
    DeltaI=np.zeros(np.size(xi))
    if np.size(xi) > 1:
        for i in range(0,np.size(xi)):
            DeltaI[i]=-get_I(q,Gamma,xi[i],0.99999999*xid) # minus sign since limits are reversed
    else:
        DeltaI=-get_I(q,Gamma,xi,0.99999999*xid)
    
    S=(n+0.5)*((b-w*L*xi**2.)/(Y*xi))-b*(1./xi-1./xid)-n-w*(DeltaI)-0.5*(hp/hp_d)**2.*((b-w*L*xi**2.)/(Y*xi))**(2.*n+1.)
    
    return S
